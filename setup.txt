docker run -t -i --net=bridge gizm00/pycon2016_lamp /bin/bash

restart apache: /etc/init.d/apache2 restart
get LAMP container ip, first line of: ifconfig | grep 'inet addr'

git pycon2016 to local dir. cd to that dir

start ipython instance
docker run --rm -it --net=bridge -p 8888:8888 -v "$(pwd):/notebooks" gizm00/pycon2016

once started launch the notebook in the background:
source docker/start-notebook.sh

test with 
req = requests.get('http://172.17.0.2')

1. check $docker network inspect bridge and confirm containers are on the same network
